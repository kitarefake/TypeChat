order -> item*
item -> pizza | salad | beer
pizza -> { type = pizza pizza-props }
pizza-props -> size? toppings?
size -> ; (small | medium | large | extra large)
toppings -> ; toppings = topping-list
topping-list -> topping (, topping)*
topping -> pepperoni | sausage | mushrooms | basil | extra cheese | extra sauce | anchovies | pineapple | olives | arugula | Canadian bacon
salad -> { type = salad salad-props }
salad-props -> saladSize? style? removed?
saladSize -> ; (whole | half)
style -> ; style = (Garden | Greek)
removed -> ; removed = removed-list
removed-list -> removed-item (, removed-item)*
removed-item -> red onions | croutons
beer -> { type = beer beer-props }
beer-props -> kind quantity?
kind -> ; kind = (Sierra Nevada Pale Ale | Mack & Jacks)
quantity -> ; quantity = number
